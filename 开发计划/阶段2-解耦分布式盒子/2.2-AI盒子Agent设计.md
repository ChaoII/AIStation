# 阶段2：解耦分布式盒子 — AI 盒子 Agent 设计

## Agent 概述

运行在每个 AI 盒子上的轻量级 Python 服务，负责：
1. 向 Web 主站注册盒子信息
2. 定期发送心跳，轮询待执行命令
3. 管理本地推理 Worker 进程
4. 持久化任务配置以支持重启恢复

## agent.py — 核心逻辑

```python
class BoxAgent:
    def __init__(self, config: dict):
        self.config = config
        self.token: str | None = None         # Web 下发 JWT
        self.registered = False
        self.task_manager = TaskManager(config)
        self.local_store = LocalStore(config.tasks_dir)

    async def register(self) -> bool:
        """向 Web 主站注册（含重试逻辑）"""
        payload = {
            "box_id": self.config.box_id,
            "name": self.config.box_name,
            "secret_key": self.config.secret_key,
            "ip_address": get_local_ip(),
            "agent_port": 9510,
            "api_version": "1.0.0",
            "gpu_info": self.config.gpu,
            "cpu_info": self.config.cpu,
            "memory_total": self.config.memory_total_mb,
            "max_tasks": self.config.max_tasks,
        }
        resp = await httpx.post(self.config.register_url, json=payload, timeout=10)
        data = resp.json()
        if data["code"] == 200:
            self.token = data["data"]["token"]
            self.registered = True
            return True
        return False

    async def send_heartbeat(self) -> list[dict]:
        """发送心跳，返回待执行命令列表"""
        tasks_status = self.task_manager.get_all_status()
        resource_usage = self._get_resource_usage()
        payload = {
            "box_id": self.config.box_id,
            "token": self.token,
            "timestamp": datetime.utcnow().isoformat(),
            "tasks": tasks_status,
            "resource_usage": resource_usage,
        }
        resp = await httpx.post(
            self.config.heartbeat_url, json=payload,
            timeout=15,
            headers={"Authorization": f"Bearer {self.token}"}
        )
        return resp.json().get("data", {}).get("commands", [])

    async def execute_command(self, cmd: dict):
        """执行单条命令"""
        action = cmd["action"]
        task_id = cmd["task_id"]
        if action == "start_task":
            config = cmd["config"]  # 全量任务配置
            self.local_store.save(task_id, config)
            self.task_manager.start(task_id, config)
        elif action == "stop_task":
            self.task_manager.stop(task_id)
            self.local_store.remove(task_id)
        elif action == "restart_task":
            self.task_manager.stop(task_id)
            config = self.local_store.load(task_id)
            if config:
                self.task_manager.start(task_id, config)

    async def main_loop(self):
        """主循环"""
        while not self.registered:
            await self.register()
            await asyncio.sleep(10)

        # 恢复本地持久化任务
        for task_id, config in self.local_store.list_all():
            self.task_manager.start(task_id, config)

        while True:
            try:
                commands = await self.send_heartbeat()
                for cmd in commands:
                    await self.execute_command(cmd)
            except (httpx.ConnectError, httpx.TimeoutException, httpx.HTTPStatusError) as e:
                logger.warning(f"主站不可达: {e}，继续本地运行")
            await asyncio.sleep(self.config.heartbeat_interval)
```

class TaskManager:
    def __init__(self, config):
        self.config = config
        self._tasks: dict[int, dict] = {}
        # 同 _running_inferences 结构

    def start(self, task_id: int, task_config: dict):
        """启动推理 Worker 进程"""
        config_path = self._write_config(task_id, task_config)
        proc = subprocess.Popen(
            [self.config.worker_python, "-u", "worker.py", "--config", config_path],
            stdout=subprocess.DEVNULL,
            stderr=subprocess.PIPE,
        )
        self._tasks[task_id] = {
            "proc": proc,
            "config_path": config_path,
            "start_time": datetime.now(),
            "task_config": task_config,
        }

    def stop(self, task_id: int):
        info = self._tasks.pop(task_id, None)
        if info:
            info["proc"].send_signal(signal.SIGTERM)
            try:
                info["proc"].wait(timeout=5)
            except subprocess.TimeoutExpired:
                info["proc"].kill()
                info["proc"].wait()
            Path(info["config_path"]).unlink(missing_ok=True)

    def get_all_status(self) -> list[dict]:
        """返回所有任务状态（用于心跳上报）"""
        result = []
        for tid, info in self._tasks.items():
            proc = info["proc"]
            status = "RUNNING" if proc.poll() is None else "STOPPED"
            uptime = (datetime.now() - info["start_time"]).total_seconds()
            result.append({
                "task_id": tid,
                "local_status": status,
                "uptime_sec": int(uptime),
                "pid": proc.pid,

## main.py — 入口

```python
import asyncio
import signal
import yaml
from agent import BoxAgent

async def main():
    with open("config.yaml") as f:
        config = yaml.safe_load(f)
    agent = BoxAgent(config)

    # 优雅退出
    stop = asyncio.Event()
    for sig in (signal.SIGTERM, signal.SIGINT):
        asyncio.get_event_loop().add_signal_handler(
            sig, lambda: stop.set()
        )

    async def run():
        await agent.main_loop()

    task = asyncio.create_task(run())
    await stop.wait()
    task.cancel()
    # 停止所有 Worker
    agent.task_manager.stop_all()

if __name__ == "__main__":
    asyncio.run(main())
```
